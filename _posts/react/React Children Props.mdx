`props` 接受一个 `children` 参数，这个参数可以是一个字符串，也可以是一个数组，数组中的每一项都是一个字符串

```jsx
function App() {
  return (
    <User>
      <h1>Hello, John Doe!</h1>
    </User>
  )
}

function User({children}) {
  return children // Hello, John Doe!
}
```
